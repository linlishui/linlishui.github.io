概述#

Shell 是操作系统的命令行解释器，是用户与操作系统内核交互的接口。常见的 Shell 类型包括：

  • Bash：Linux 默认 Shell，功能丰富
  • Zsh：增强版 Shell，支持更多特性

参数#

脚本参数小知识

#!/bin/bash

# 1. 位置参数
## 脚本名: script.sh
echo "脚本名称: $0"
echo "第一个参数: $1"
echo "第二个参数: $2"
echo "第三个参数: $3"

## 使用方式: ./script.sh arg1 arg2 arg3

# 2. 特殊参数变量
echo "参数总数: $#"
echo "所有参数(作为单个字符串): $*"
echo "所有参数(作为独立字符串): $@"
echo "脚本进程ID: $$"
echo "上一个命令退出状态: $?"

# 3. 参数检查
if [ $# -eq 0 ]; then
    echo "用法: $0 <文件名> [选项]"
    exit 1
fi

if [ $# -lt 2 ]; then
    echo "错误: 需要至少2个参数"
    exit 1
fi

# 4. 参数默认值
filename=${1:-"default.txt"}
action=${2:-"read"}
verbose=${3:-"false"}

echo "文件名: $filename"
echo "操作: $action"
echo "详细模式: $verbose"

循环结构#

算数运算解释

# 为什么使用 ((count++))？
count=1

# 方法1：((表达式)) - 算术扩展，支持C风格运算
((count++))     # count = count + 1
((count += 2))  # count = count + 2
((count *= 3))  # count = count * 3

# 方法2：$((表达式)) - 算术替换
count=$((count + 1))
new_value=$((count * 2))

# 方法3：let命令
let count++
let "count += 5"

# 方法4：expr命令（较老的方式）
count=$(expr $count + 1)

循环实例详解

# for循环的多种形式
# 1. 数字序列
for i in {1..10}; do
    echo "数字: $i"
done

# 2. 带步长的序列
for i in {1..20..2}; do  # 1到20，步长为2
    echo "奇数: $i"
done

# 3. 数组遍历
fruits=("苹果" "香蕉" "橙子")
for fruit in "${fruits[@]}"; do
    echo "水果: $fruit"
done

# 4. 文件遍历
for file in *.txt; do
    if [ -f "$file" ]; then
        echo "处理文件: $file"
    fi
done

# C风格for循环
for ((i=1; i<=5; i++)); do
    echo "C风格循环: $i"
done

带参数和返回值的函数

# 函数定义方式1
function calculate_sum() {
    local num1=$1  # 第一个参数
    local num2=$2  # 第二个参数
    local result=$((num1 + num2))
    
    echo "计算 $num1 + $num2 = $result"
    return $result  # 返回值（0-255）
}

# 函数定义方式2
multiply() {
    local a=$1
    local b=$2
    echo $((a * b))  # 通过echo返回结果
}

# 函数调用和返回值处理
calculate_sum 10 20
sum_result=$?  # 获取return的返回值

product_result=$(multiply 6 7)  # 获取echo的输出
echo "乘积结果: $product_result"

# 处理多个参数
process_files() {
    echo "参数个数: $#"
    echo "所有参数: $@"
    echo "脚本名: $0"
    
    for arg in "$@"; do
        echo "处理参数: $arg"
    done
}

process_files file1.txt file2.txt file3.txt

文件操作#

文件读取的多种方式

# 方法1：while read逐行读取
read_file_line_by_line() {
    local filename=$1
    local line_number=1
    
    if [ ! -f "$filename" ]; then
        echo "错误: 文件 $filename 不存在"
        return 1
    fi
    
    while IFS= read -r line; do
        echo "第${line_number}行: $line"
        ((line_number++))
    done < "$filename"
}

# 方法2：读取到数组
read_file_to_array() {
    local filename=$1
    local -a file_lines=()
    
    while IFS= read -r line; do
        file_lines+=("$line")
    done < "$filename"
    
    echo "文件共 ${#file_lines[@]} 行"
    for i in "${!file_lines[@]}"; do
        echo "行$((i+1)): ${file_lines[i]}"
    done
}

# 方法3：处理CSV文件
process_csv() {
    local csv_file=$1
    local line_num=0
    
    while IFS=',' read -r col1 col2 col3; do
        ((line_num++))
        if [ $line_num -eq 1 ]; then
            echo "表头: $col1 | $col2 | $col3"
        else
            echo "数据: $col1 | $col2 | $col3"
        fi
    done < "$csv_file"
}

# 使用示例
echo -e "姓名,年龄,城市\n张三,25,北京\n李四,30,上海" > sample.csv
process_csv "sample.csv"

文件操作实用函数

# 安全的文件操作函数
safe_file_operation() {
    local source_file=$1
    local operation=$2
    
    # 检查文件是否存在
    if [ ! -f "$source_file" ]; then
        echo "错误: 源文件 $source_file 不存在"
        return 1
    fi
    
    # 检查文件是否可读
    if [ ! -r "$source_file" ]; then
        echo "错误: 文件 $source_file 不可读"
        return 2
    fi
    
    case $operation in
        "backup")
            local backup_file="${source_file}.bak.$(date +%Y%m%d)"
            cp "$source_file" "$backup_file"
            echo "备份完成: $backup_file"
            ;;
        "size")
            local file_size=$(stat -f%z "$source_file" 2>/dev/null || stat -c%s "$source_file" 2>/dev/null)
            echo "文件大小: $file_size 字节"
            ;;
        *)
            echo "不支持的操作: $operation"
            return 3
            ;;
    esac
}

# 使用示例
safe_file_operation "sample.csv" "backup"
safe_file_operation "sample.csv" "size"
